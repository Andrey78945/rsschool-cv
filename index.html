Code example: class DepthCalculator with method calculateDepth that calculates depth of nested array

                  
                    class DepthCalculator {
                      calculateDepth(arr) {
                        if (arr.every(el => !(Array.isArray(el)))) {
                          return 1;
                        }
                        let res = new Array;
                        arr.forEach(element => {
                          if (Array.isArray(element)) {
                            res.push(1 + this.calculateDepth(element));
                          }
                        });
                        return res.sort((a, b) => b - a)[0];
                      }
                    }
